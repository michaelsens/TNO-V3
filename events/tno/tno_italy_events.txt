tno_italy_events.1 = { # Levent becomes isreal 
	type = country_event
    placement = ROOT
 
    title = tno_italy_events.1

		texture  = "gfx/event_pictures/super_pictures/lasthope.dds"

        name = tno_italy_events.1.a

			culture = cu:ashkenazi
			ruler = yes
			interest_group = ig_landowners
			ideology = ideology_fascist
			traits = {
				tactful
				bigoted
				expert_political_operator
			}
		}

		create_country = { 
			tag = EOT
			origin = c:ISR
	 
			state = s:STATE_TRANSJORDAN.region_state:ISR
		
		 
		   
			
			on_created = {
				set_immune_to_revolutions = yes
				}
		} 
		trigger_event = { id = tno_italy_events.3 days = 0 popup = yes }

	}

	option = { # levant becomes palistine and isreal, partition
        name = tno_italy_events.1.b

		province = p:x60D060
		province = p:xC9CBC0
  
      
			on_created = {
            set_immune_to_revolutions = yes
	        }
	    } 

		create_country = { 
			tag = EOT
			origin = c:ISR
	 
			state = s:STATE_TRANSJORDAN.region_state:ISR
		
		 
		   
			
			on_created = {
				set_immune_to_revolutions = yes
				}
		} 


		create_country = { 
				tag = LEB
				origin = c:ISR
		 
				state = s:STATE_LEBANON.region_state:ISR
			
			 
			   
				
				on_created = {
					set_immune_to_revolutions = yes
					}
		} 
		

	}



	


	option = { # Communist revolution
        name = tno_italy_events.1.c

		
		activate_law = law_type:law_council_republic
		activate_law = law_type:law_command_economy

		create_country = { 
			tag = EOT
			origin = c:PAL
	 
			state = s:STATE_TRANSJORDAN.region_state:PAL
		
		 
		   
			
			on_created = {
				set_immune_to_revolutions = yes
				}
		} 

		trigger_event = { id = tno_italy_events.2 days = 0 popup = yes }


		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ITA
					 type = reichskommissariat
				}
			}
			remove_diplomatic_pact = {
				country = c:ITA
				type = reichskommissariat
			}
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ITA
					 type = customs_union
				}
			}
			remove_diplomatic_pact = {
				country = c:ITA
				type = customs_union
			}
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ITA
					 type = puppet
				}
			}
			remove_diplomatic_pact = {
				country = c:ITA
				type = puppet
		   }
		   
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ITA
					 type = dominion
				}
			}
			remove_diplomatic_pact = {
				country = c:ITA
				type = dominion
		   }
		   
		}
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:ITA
					 type = protectorate
				}
			}
			remove_diplomatic_pact = {
				country = c:ITA
				type = protectorate
		   }
		   
		}

		
		


	}

}
