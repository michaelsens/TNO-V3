tno_usa_segregation_events.1 = { # civil rights act
    type = country_event
    placement = ROOT

	title = tno_usa_segregation_events.1
	desc = tno_usa_segregation_events.1.d
    flavor = tno_usa_segregation_events.1.f	

	event_image = {
		texture  = "gfx/event_pictures/super_pictures/seg.6.dds"
	}

    icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	trigger = {
		this ?= c:USA
	}	
	
	duration = 1

	option = { # pass civil rights act
		default_option = yes
        name = tno_usa_segregation_events.1.a
		activate_law = law_type:law_multicultural

        add_journal_entry = { type = je_southernriots }

        add_modifier = southern_riots

        trigger_event = { id = tno_usa_segregation_events.8 } #triggers southern riots across all states

        add_loyalists = {
            value = 0.20
            culture = cu:afro_american
        }
        add_radicals = {
            value = 0.05
            culture = cu:yankee
        }
	}

    option = { # Veto civil rights act
		name = tno_usa_segregation_events.1.b

        add_journal_entry = { type = je_northernriots }

        add_modifier = northern_riots

        trigger_event = { id = tno_usa_segregation_events.12 } #triggers northern riots across all states

        add_radicals = {
            value = 0.25
            culture = cu:afro_american
        }
        add_loyalists = {
            value = 0.08
            culture = cu:yankee
        }
		
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
}
