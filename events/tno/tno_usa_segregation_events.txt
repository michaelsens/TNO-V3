namespace = tno_usa_segregation_events

tno_usa_segregation_events.1 = { # civil rights act
    type = country_event
    placement = ROOT

	title = tno_usa_segregation_events.1
	desc = tno_usa_segregation_events.1.d
    flavor = tno_usa_segregation_events.1.f	

	event_image = {
		texture  = "gfx/event_pictures/super_pictures/liberty.dds"
	}

    icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	trigger = {
		this ?= c:USA
	}	
	
	duration = 1

	option = { # pass civil rights act
		name = tno_usa_segregation_events.1.a
		activate_law = law_type:law_multicultural

        add_journal_entry = { type = je_southernriots }

        add_loyalists = {
            value = 0.20
            culture = cu:afro_american
        }
        add_radicals = {
            value = 0.05
            culture = cu:yankee
        }
	}

    option = { # Veto civil rights act
		name = tno_usa_segregation_events.1.b

        add_radicals = {
            value = 0.30
            culture = cu:afro_american
        }
        add_loyalists = {
            value = 0.10
            culture = cu:yankee
        }
        add_radicals = {
            value = 0.03
            culture = cu:yankee
        }
		
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
}
