    flavor = tno_RK_events.2.f	

	event_image = { 
		texture  = "gfx/event_pictures/event_pictures/arms_raid.dds"
	}

    icon = "gfx/interface/icons/tno_event_icons/bicorne_jim_crow_protest.2.dds"
	
	trigger = {
		this ?= c:RKO
	}	

    immediate = {
        random_scope_state = {
            save_scope_as = arms_raid_state
            limit = {
              OR = {
                state_region = s:STATE_ESTONIA
                state_region = s:STATE_LATVIA
                state_region = s:STATE_LITHUANIA
				state_region = s:STATE_INGRIA
				state_region = s:STATE_MINSK
				state_region = s:STATE_BREST
				state_region = s:STATE_ORSHA
              }
            }
        }

		change_variable = {
            name = chaos_ost_timer
            add = 5
        }

    }
	
	duration = 1

	option = {
		default_option = yes
        name = tno_RK_events.2.a

        scope:arms_raid_state = {
            add_radicals_in_state = {
                value = 0.02
            }
        }
		scope:arms_raid_state.state_region = {
			add_devastation = 35
  		}

		  add_modifier = {
            name = harsh_reprisals
            months = 12
        } 
        
	}
    option = {
		default_option = yes
        name = tno_RK_events.2.b

		scope:arms_raid_state.state_region = {
			add_devastation = 10
  		}
