namespace = tno_dynamic_puppets


tno_dynamic_puppets.1 = {
    type = country_event
    hidden = yes



    trigger = {
        exists = c:WAD
        ROOT = c:ITA

        c:WAD = {
            has_diplomatic_pact = {
				who = c:ITA
                type = protectorate
            }

        }



    }

    immediate = {
        remove_diplomatic_pact = {
            country = c:WAD
            type = protectorate
        }
		create_diplomatic_pact = {
            country = c:WAD
            type = millgov
        }
    }
}
