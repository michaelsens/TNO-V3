	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				has_variable = take_share_target_state
			}
			save_scope_as = take_share_target_state

			limit = {
				has_variable = fi_identification
			}
			trigger_event = { id = fi_dividend_collector.5 }
		}
	}

	option = { 
		name = fi_share_taking_executer.1.a
		default_option = yes

		every_country = {
			limit = {
				has_variable = fi_identification
				is_target_in_variable_list = {
					name = invested_states
					target = scope:take_share_target_state
				}
			}
			change_relations = {
				country = root
				value = scope:take_share_target_state.state_total_number_capped_10_times_3
			}
			scope:take_share_target_state.state_region = {
				add_claim = prev
			}
		}

		change_infamy = scope:take_share_target_state.state_total_number_capped_10
		hidden_effect = { trigger_event = { id = fi_share_taking_executer.2 } }
	}

	option = { 
		name = fi_share_taking_executer.1.b
		scope:take_share_target_state = {
			remove_variable = take_share_target_state
		}
	}
}

fi_share_taking_executer.41 = {
	type = country_event
	hidden = yes

	trigger = {
		has_variable = total_invested_building
		var:total_invested_building > 30
		has_technology_researched = watertube_boiler
		has_technology_researched = dynamite
		any_in_list = {
			variable = invested_states
			owner = {
				NOT = {
					has_variable = mine_exclusive
					OR = {
						has_technology_researched = watertube_boiler
						has_technology_researched = dynamite
					}
				}
			}
		}
	}

	immediate = {
		clear_variable_list = invested_states
		root = { save_scope_as = investor_country }

			fi_add_list = yes
		}
		if = {
			limit = {
				any_in_list = {
					variable = invested_states
					owner = {
						NOT = {
							has_variable = mine_exclusive
							OR = {
								has_technology_researched = watertube_boiler
								has_technology_researched = dynamite
							}
						}
					}
				}
			}
			set_variable = tech_target_country
			random_in_list = {
				variable = invested_states
				limit = {
					owner = {
						NOT = {
							has_variable = mine_exclusive
							OR = {
								has_technology_researched = watertube_boiler
								has_technology_researched = dynamite
							}
						}
					}
				}
				set_variable = tech_target_state
				owner = { trigger_event = { id = fi_share_taking_executer.5 } }
			}
		}
	}
}
