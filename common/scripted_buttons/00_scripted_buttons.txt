je_oil_crisis_oil_supply_button_button_d = {  # Nationalize the Oil Production
	name = "je_oil_crisis_oil_supply_name_4"
	desc = "je_oil_crisis_oil_supply_name_4_desc"

	visible = { always = yes }

	possible = {
		scope:journal_entry = {
		NOT = {
			custom_tooltip = {
				text = oil_supply_nationalization_tt
				has_variable = je_oil_supply_nationalization
			}
			}
		}
	}

	effect = {
		change_variable = {
			name = oil_crisis_oil_supply_var
			add = 500
		}
		hidden_effect= {
			scope:journal_entry = {
				set_variable = { name = je_oil_supply_nationalization }
			}
		}
		add_technology_researched = oil_crisis_nationalization
		activate_production_method = { 
			building_type = building_oil_rig
			production_method = pm_tno_oil_government_run_industry
		}	
}
}

je_oil_crisis_oil_supply_button_button_e = {  # Research new Technologies
	name = "je_oil_crisis_oil_supply_name_5" 
	desc = "je_oil_crisis_oil_supply_name_5_desc"

	visible = { always = yes }

	possible = {
		NOT = {
			OR = {
			  has_technology_researched = combustion_engine
			  has_technology_researched = nuclear_times
			}
		scope:journal_entry = {
		  custom_tooltip = {
			text = oil_supply_tt
			has_variable = je_oil_supply_cooldown
		}
		}
	}
	}

	effect = {
		change_variable = {
			name = oil_crisis_oil_supply_var
			add = 50
		}
		scope:journal_entry = {
			custom_tooltip = cooldown_three_months_oil_supply_tt
			set_variable = { name = je_oil_supply_cooldown months = 3 }
		}	
		if = {
         limit = {
			NOT = {
				has_technology_researched = combustion_engine
			  }
			 }
			 add_technology_progress = { 
				progress = 5000 technology = combustion_engine
			}
		}
		if = {
			limit = {
			   NOT = {
				   has_technology_researched = nuclear_times
				 }
				}
				add_technology_progress = { 
				   progress = 5000 technology = nuclear_times
			   }
		   }
	}
}
