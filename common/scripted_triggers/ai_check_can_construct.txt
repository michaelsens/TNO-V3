ai_check_can_construct = {
    any_scope_state = {
        market_access > 0.95
        OR = {
            AND = {
                NOT = { 
                    any_scope_building = {
                        is_building_type = $BUILDING$
                    } 
                }
                $BUILDING$_condition = yes
                $BUILDING$_total_queued < 1
            }
            AND = {
                any_scope_building = {
                    is_building_type = $BUILDING$
                    occupancy > 0.95
                    cash_reserves_ratio > 0.95
                    $BUILDING$_can_queue = yes
                } 
            }
        }
    }
}
