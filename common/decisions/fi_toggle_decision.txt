fi_toggle_on_decision = {
	is_shown = {
		is_player = yes
        NOT = { has_variable = fi_toggle_on}
        capital = { owner = root }
	}

	possible = {
		NOT = { has_variable = fi_toggle_on}
        
	}

	when_taken = {
		
		custom_tooltip = {
            text = fi_toggle_on
            set_variable = fi_toggle_on
        }
	}
}

fi_toggle_off_decision = {
    is_shown = {
        is_player = yes
        has_variable = fi_toggle_on
        capital = { owner = root }
    }

    possible = {
        has_variable = fi_toggle_on
        
    }

    when_taken = {
        
        custom_tooltip = {
            text = fi_toggle_off
            remove_variable = fi_toggle_on
        }
    }
}
