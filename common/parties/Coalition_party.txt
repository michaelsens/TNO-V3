# Bastion of the Landowners

Coalition_party = {
	color = { 108 64 179 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = party_RDCS
				trigger = {
					exists = c:USA
					THIS = c:USA
				}
			}
		}
	}
	icon = {
		default = "/gfx/interface/icons/political_parties/conservatism.dds"
	}
	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		always = yes
	}
	available_for_interest_group = {
		owner = {
			country_has_voting_franchise = yes
			ig:ig_Moderates = {
				is_marginal = no
			}			
		}
		# unions can never be conservative
		NOT = { is_interest_group_type = ig_trade_unions }
		NOT = { is_interest_group_type = ig_Communists }
		NOT = { is_interest_group_type = ig_Progressives }
		NOT = { is_interest_group_type = ig_Socialist }		
		NOT = { is_interest_group_type = ig_Anarchists }	
		NOT = { is_interest_group_type = ig_trade_unions }
		NOT = { is_interest_group_type = ig_Communists }
		NOT = { is_interest_group_type = ig_Progressives }
		NOT = { is_interest_group_type = ig_Socialist }		
		NOT = { is_interest_group_type = ig_Anarchists }		
		
		OR = {
	     	NOT = {
	         	owner = {
	         		has_variable = only_valid_party
	         	}
	        }
	        owner.var:only_valid_party = {
	        	is_party_type = Coalition_party
	        }
	    }
	}
	
	# Root - IG considering joining
	# scope:number - the number of IGs that would join this party before the root IG
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}
		if = { # socialists and radicals won't join conservative parties
			limit = {
				leader = { 
					OR = {
						has_ideology = ideology:ideology_communist
						has_ideology = ideology:ideology_vanguardist
						has_ideology = ideology:ideology_anarchist
						has_ideology = ideology:ideology_radical
						has_ideology = ideology:ideology_atheist
					}
				}
			}
			add = {
				value = -20
				desc = "from_leftist"
			}
		}
		if = { # conservatives love conservativism
			limit = {
				leader = {
					OR = {
						has_ideology = ideology:ideology_traditionalist
						has_ideology = ideology:ideology_abolitionist
						has_ideology = ideology:ideology_republican_leader
						has_ideology = ideology:ideology_moderate
						has_ideology = ideology:ideology_pacifist
					}
				}
			}
			add = {
				value = 5
				desc = "from_conservative_leader"
			}
		}
		if = { # RDC loves centrists
			limit = {
				OR = {
					has_ideology = ideology:ideology_republican
					has_ideology = ideology:ideology_individualist
					has_ideology = ideology:ideology_patriarchal
					has_ideology = ideology:ideology_liberal
					has_ideology = ideology:ideology_egalitarian
				}
			}
			add = {
				value = 5
				desc = "from_conservative_ig"
			}
		}
		if = { # Goldwater wing of the RDC
			limit = {
				OR = {
					has_ideology = ideology:ideology_laissez_faire
				}
			}
			add = {
				value = 5
				desc = "from_conservative_ig"
			}
		}		
		
			if = { # a party for powerful farmers
			limit = {
				is_interest_group_type = ig_Conservatives
				is_powerful = yes
			}
			add = {
				value = 10
			}
		}
			if = { # a party for powerful farmers
			limit = {
				is_interest_group_type = ig_Liberals
				is_powerful = yes
			}
			add = {
				value = 10
			}
		}	
		if = { # a party for powerful farmers
			limit = {
				is_interest_group_type = ig_Moderates
				is_powerful = yes
			}
			add = {
				value = 10
			}
		}	
		if = { # a party for powerful farmers
			limit = {
				is_interest_group_type = ig_Libertarians
				is_powerful = yes
			}
			add = {
				value = 10
			}
		}		

		if = {
			limit = {
				scope:number = 1
				is_powerful = no
				is_marginal = no
			}
			multiply = {
				value = 2
				desc = "from_influential_supporting_member"
			}
		}
		else_if = {
			limit = {
				scope:number = 2
			}
			multiply = {
				value = 0.5
				desc = "from_two_prior_members"
			}
		}
		else_if = {
			limit = {
				scope:number = 3
			}
			multiply = {
				value = 0.25
				desc = "from_three_prior_members"
			}
		}
		else_if = {
			limit = {
				scope:number > 3
			}
			multiply = {
				value = 0.1
				desc = "from_more_than_three_prior_members"
			}
		}
		if = { # anti-orleanist
			limit = {
				leader = {
					OR = {
						has_ideology = ideology:ideology_bonapartist
						has_ideology = ideology:ideology_legitimist
					}
				}
				owner = {
					any_interest_group = {
						has_party = yes
						party = {
							is_party_type = conservative_party
						}
						is_powerful = yes
						leader = {
							has_ideology = ideology:ideology_orleanist
						}
					}
				}
			}
			add = {
				value = -5
				desc = "from_french_monarchist_rival"
			}
		}
		if = { # anti-legitimist
			limit = {
				leader = {
					OR = {
						has_ideology = ideology:ideology_bonapartist
						has_ideology = ideology:ideology_orleanist
					}
				}
				owner = {
					any_interest_group = {
						has_party = yes
						party = {
							is_party_type = conservative_party
						}
						is_powerful = yes
						leader = {
							has_ideology = ideology:ideology_legitimist
						}
					}
				}
			}
			add = {
				value = -5
				desc = "from_french_monarchist_rival"
			}
		}
		if = { # anti-bonapartist
			limit = {
				leader = {
					OR = {
						has_ideology = ideology:ideology_legitimist
						has_ideology = ideology:ideology_orleanist
					}
				}
				owner = {
					any_interest_group = {
						has_party = yes
						party = {
							is_party_type = conservative_party
						}
						is_powerful = yes
						leader = {
							has_ideology = ideology:ideology_bonapartist
						}
					}
				}
			}
			add = {
				value = -5
				desc = "from_french_monarchist_rival"
			}
		}
		if = {
			limit = {
				owner = {
	         		has_variable = only_valid_party
	         	}
	         	NOT = {
			        owner.var:only_valid_party = {
			        	is_party_type = conservative_party
			        }
			    }
			}
			multiply = {
				value = 0
			}
		}
		if = {
			limit = {
				leader = {
					has_ideology = ideology:ideology_jacksonian_democrat
				}
			}
			add = {
				value = 1000
				desc = "from_jacksonian_democrat"
			}
		}
		if = {
			limit = {
				owner = {
	         		has_variable = only_valid_party
	         	}
				owner.var:only_valid_party = {
					is_party_type = conservative_party
				}
	         	owner.ruler = {
					this.interest_group = prev
				}
			}
			add = {
				value = 1000
				desc = "from_only_legal_party"
			}
		}
		else_if = {
			limit = {
				owner = {
	         		has_variable = only_valid_party
	         	}
	         	 owner.var:only_valid_party = {
		        	is_party_type = conservative_party
		        }
			}
			add = {
				value = 3
				desc = "from_only_legal_party"
			}
		}
	}
}
