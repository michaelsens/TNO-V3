fi_expand_disabled = {
	type = country
	random_valid = yes

	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = investor_country
			}
			OR = {
				NOT = { scope:investor_country = { market_capital = { owner = scope:investor_country } } }
				check_investable = no
			}
		}
		localization_key = "FI_EXPAND_DISABLED_TOOLTIP2"
	}
	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = investor_country
			}
			has_variable = mine_exclusive
			scope:investor_country = { market_capital = { owner = scope:investor_country } }
			check_investable = yes	
		}
		localization_key = "FI_EXPAND_DISABLED_MINE_EXCLUSIVE"
	}
}

fi_expand_treaty = {
	type = country
	random_valid = yes

	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = player
			}
			scope:player = { has_treaty_port_in_country = root }
		}
		localization_key = "YES_CHECK"
	}
	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = player
			}
			NOT = { scope:player = { has_treaty_port_in_country = root } }
		}
		localization_key = "NO_CHECK"
	}
}
fi_expand_free = {
	type = country
	random_valid = yes

	text = {
		trigger = {
			has_law = law_type:law_free_trade
		}
		localization_key = "YES_CHECK"
	}
	text = {
		trigger = {
			NOT = { has_law = law_type:law_free_trade }
		}
		localization_key = "NO_CHECK"
	}
}
fi_expand_custom = {

	}
}
fi_expand_trade = {
	type = country
	random_valid = yes

	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = player
			}
			market_capital.owner = {
				has_diplomatic_pact = {
					who = scope:player.market_capital.owner
					type = trade_agreement
				}
			}
		}
		localization_key = "YES_CHECK"
	}
	text = {
		trigger = {
			any_country = {
				is_player = yes
				save_temporary_scope_as = player
			}
			NOT = { 
				market_capital.owner = {
					has_diplomatic_pact = {
						who = scope:player.market_capital.owner
						type = trade_agreement
					}
				}
			}
		}
		localization_key = "NO_CHECK"
	}
}
