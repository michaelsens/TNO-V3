je_jimcrow = { # Jim crow itself not civil rights act

    icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

    
    
    complete = {
        OR = { 
            has_law = law_type:law_multicultural
        }
    }

    on_complete = {
        remove_modifier = jim_crow

        add_radicals = {
            value = 0.05
            culture = cu:yankee
        }
        add_loyalists = {
            value = 0.05
            culture = cu:afro_american
        }
    }

    on_monthly_pulse = {
        random_events = {
            10 = 0
            15 = tno_usa_segregation_events.2
            15 = tno_usa_segregation_events.3
            15 = tno_usa_segregation_events.4
            15 = tno_usa_segregation_events.5
            15 = tno_usa_segregation_events.6
        }
    }
    
    weight = 100
    
    should_be_pinned_by_default = no
    
}

je_civil_rights_act = { # civil rights act

    icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

    immediate = {

		set_variable = {
			name = civil_rights_timer
			value = 0
		}
	}
    
    complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

    on_complete = {
        trigger_event = { id = tno_usa_segregation_events.1 popup = yes }
    }

    on_monthly_pulse = {
        effect = {
			change_variable = {
				name = civil_rights_timer
				add = 1
			}
		}
    }

    current_value = {
		value = root.var:civil_rights_timer
	}

    goal_add_value = {
		value = 24
	}

    progressbar = yes
    
    weight = 100
    
    should_be_pinned_by_default = yes
    
}

		set_variable = {
			name = civil_rights_timer

		}

			}

		value = root.var:civil_rights_timer
	}

    goal_add_value = {
		value = 6
	}

    progressbar = yes
    
    weight = 100
    
    should_be_pinned_by_default = yes
    
}
