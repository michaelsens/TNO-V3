je_tno_insergency_ost = { # ostland insurgency

    icon = "gfx/interface/icons/new_event_icons/event_ost_insur.dds"

    scripted_button = je_insur_ost_a
    scripted_button = je_insur_ost_B

    immediate = {

		set_variable = {
			name = chaos_ost_timer
			value = 0
		}
	}
    
    complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

    on_complete = {
        trigger_event = { id = tno_usa_segregation_events.1 popup = yes }
    }

    on_monthly_pulse = {
        effect = {
			change_variable = {
				name = chaos_ost_timer
				add = -1
			}
		}
		random_events = {
            6 = 0
            3 = tno_usa_segregation_events.2
            3 = tno_usa_segregation_events.3
            3 = tno_usa_segregation_events.4
            3 = tno_usa_segregation_events.5
            3 = tno_usa_segregation_events.6
        }
    }

    current_value = {
		value = root.var:chaos_ost_timer
	}

    on_timeout = {
		trigger_event = { id = tno_usa_segregation_events.15 popup = yes }
	}

    goal_add_value = {
		value = 100
	}

    progressbar = yes
    
    weight = 100

    should_be_pinned_by_default = yes
    
}
