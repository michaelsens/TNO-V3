je_ru_tno_irk_paris_of_siberia = { 

    #icon = "gfx/interface/icons/invention_icons/urbanization.dds"
    icon = "gfx/interface/icons/tno_event_icons/paris_of_siberia.dds"
    group = je_group_technology

    immediate = {
        ruler = {
            save_scope_as = irkutsk_leader
        }
        random_scope_state = {
            limit = {
                state_region = s:STATE_IRKUTSK
            }
            save_scope_as = irkutsk_state_scope
        }
    }
    
    complete = {
        custom_tooltip = {
            text = je_paris_of_siberia_rq
            any_scope_state = {
                state_region = s:STATE_IRKUTSK
                state_has_building_group_levels = {
                    type = bg_technology
                    value  >= 2
                }
                has_decree = decree_encourage_manufacturing_industry
                total_urbanization >= 10
            }
        }
    }

    on_complete = {
        custom_tooltip = {
            text = je_paris_of_siberia_complete
            scope:irkutsk_state_scope.state_region = {
                add_state_trait = state_trait_paris_of_siberia
            }
        }
        if = {
             limit = {
                scope:irkutsk_state_scope = {
                    state_has_building_group_levels = {
                        type = bg_bureaucracy
                        value  >= 4
                    }
                }
            }

            custom_tooltip = {
                text = je_paris_of_siberia_complete_extra
                add_modifier = {
                    name = the_irkutsk_model
                }
            }
        }
    }

    current_value = {

    goal_add_value = {
        
    }

    progressbar = no

    should_be_pinned_by_default = yes
    
}
