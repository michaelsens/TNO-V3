je_tno_gerpowerstruggle_goreing = { # goreings JE

    icon = "gfx/interface/icons/new_event_icons/event_jimcrow.dds"

    
    
    immediate = {

		set_variable = {
            
			name = goreing_ger_struggle_timer
			value = 0
        }
	}
    
    complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}

	}

    on_complete = {
        trigger_event = { id = tno_germany_events.5 popup = yes }
    }

    current_value = {
        
		value = root.var:goreing_ger_struggle_timer
	}

    goal_add_value = {
        
		value = 150
	}

    progressbar = yes
    
    weight = 30
    
    should_be_pinned_by_default = yes
    
}
