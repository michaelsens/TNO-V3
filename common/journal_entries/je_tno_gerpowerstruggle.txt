    icon = "gfx/interface/icons/new_event_icons/hitler_sick.dds"

    invalid = {
        OR = {
            
        
            has_variable = landowner_dub
            
            has_variable = speer_dub

            has_variable = goreing_dub

            has_variable = ss_dub
            
        }
        
    }
    
    immediate = {

		set_variable = {
            
			name = hitler_ger_struggle_timer
			value = 0
        }
	}

	status_desc = {
	
        first_valid = {
            triggered_desc = {
                desc = "Hitler is unwell"
                trigger = {
                    has_variable = hitler_unwell
                    NOT = {
                        has_variable = hitler_coma
                    }
                }
            }
            triggered_desc = {
                desc = custom_hitler_tt
				trigger = {
                    has_variable = hitler_coma
                    
                }
            }
        }
    }   

	on_weekly_pulse = {
		effect = {
		change_variable = {
            name = hitler_ger_struggle_timer
            add = 2
        }
		}
	}
    
    complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}

	}

    on_complete = {
        trigger_event = { id = tno_germany_events.17 popup = yes }
       
        set_variable = {
            
			name = ss_dub
			value = 0
        }
    }

    current_value = {
        
		value = root.var:hitler_ger_struggle_timer
	}

    goal_add_value = {
        
		value = 120
	}

   
    
    weight = 190
    
    should_be_pinned_by_default = yes
    
}
