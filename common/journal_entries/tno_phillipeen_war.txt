je_tno_phillipeen_ofn = { 

    icon = "gfx/interface/icons/new_event_icons/phill.dds"
    group = je_group_historical_content
    scripted_button = je_malaya_ofn_a
    scripted_button = je_malaya_ofn_b
    scripted_button = je_malaya_ofn_c

	group = je_group_internal_affairs


    invalid = {
        c:RNG = {
            is_in_war_together = no
        }
	}
    
    complete = {
       
        c:RNG = {
            owns_entire_state_region = STATE_LUZON
			owns_entire_state_region = STATE_VISAYAS
        }
		
		
	}

    fail = {
        c:PHI = {
            owns_entire_state_region = STATE_VISAYAS
			owns_entire_state_region = STATE_MINDANAO
        } 
    }

    on_fail = {
        add_modifier = {
            name = proxy_lost
            months = 250
        } 
    }

    on_complete = {
        add_modifier = {
            name = proxy_won
            months = 250
        } 
    }

    progressbar = no
    
    weight = 100

    
    should_be_pinned_by_default = yes
    
}

je_tno_blitz_countdown = {

	icon = "gfx/interface/icons/new_event_icons/phill.dds"

	group = je_group_internal_affairs
    immediate = {

		set_variable = {
			name = blitztimer
			value = 3
		}
	}
    
    complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

    on_complete = {
        trigger_event = { id = tno_philipeen_events.1 popup = yes }

        hidden_effect = {
            c:JAP = {
				add_journal_entry = {
                    type = je_tno_phillipeen_jap
                } 
			} 
		}
        c:USA = {
            add_journal_entry = {
                type = je_tno_phillipeen_ofn
            } 
        } 
    }

    on_monthly_pulse = {
        effect = {
			change_variable = {
				name = blitztimer
				add = 5
			}
		}
    }

    current_value = {
		value = root.var:blitztimer
	}

    goal_add_value = {
		value = 3 #190 its 3 just for testing
	}

    progressbar = yes
    
    weight = 100
    
    should_be_pinned_by_default = yes
	
}
