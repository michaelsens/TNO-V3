foreign_investments_food_industry = {
	scope = country
	saved_scopes = {
		invested_state
		investor_country
		invested_building
		is_right_click
    }
	is_shown = {
                scope:invested_state = {
                        NOT = { owner = scope:investor_country}
                }
        }
	is_valid = {
		scope:investor_country = {
			market_capital = { owner = scope:investor_country }
			has_technology_researched = manufacturies
		}
		scope:invested_state ={
			owner = {
				check_investable = yes
			}
		}
	}
	effect = {
		scope:investor_country = {
			set_variable = is_right_click_check
			while = {
				count = 5
				limit = { 
					has_variable = is_right_click_check
				}
				scope:invested_state = {
					fi_construct_stock_exchange = { 
						BUILDING = building_food_industry
					}
				}
				if = {
					limit = { NOT ={ exists = scope:is_right_click } }
					remove_variable = is_right_click_check
				}
			}
			remove_variable = is_right_click_check
		}
	}
}

foreign_investments_coal_mine = {
	scope = country
	saved_scopes = {
		invested_state
		investor_country
		invested_building
		is_right_click
    }

	is_shown = {
		scope:invested_state = {
			NOT = { owner = scope:investor_country}
		}
		OR = {
			AND = {
				NOT = { exists = scope:invested_building }
				scope:invested_state = { 
					building_coal_mine_total_queued < 1
				}
			}
			AND = {
				exists = scope:invested_building
				scope:invested_building = {
					is_building_type = building_coal_mine
					building_coal_mine_can_queue = yes
				}
			}
		}	
	}
	
	is_valid = {
		scope:investor_country = {
			market_capital = { owner = scope:investor_country }
		}
		scope:invested_state ={
			owner = {
				check_investable = yes
				OR = {
					NOT = { has_variable = mine_exclusive }
					var:mine_exclusive = scope:investor_country.var:fi_identification
				}
			}
		}
	}

	effect = {
		scope:investor_country = {
			set_variable = is_right_click_check
			while = {
				count = 5
				limit = { 
					has_variable = is_right_click_check
					OR = {
						AND = {
							NOT = { exists = scope:invested_building }
							scope:invested_state = { 
								building_coal_mine_total_queued < 1
							}
						}
						AND = {
							exists = scope:invested_building
							scope:invested_building = {
								is_building_type = building_coal_mine
								building_coal_mine_can_queue = yes
							}
						}
					}
				}

				scope:invested_state = {
					
					fi_construct_stock_exchange = { 
						BUILDING = building_coal_mine
					}
				}
				if = {
					limit = { NOT ={ exists = scope:is_right_click } }
					remove_variable = is_right_click_check
				}
			}
			remove_variable = is_right_click_check
		}		
	}
}

	scope = country
	saved_scopes = {
		invested_state
		investor_country
		invested_building
		is_right_click
    }

	is_shown = {
		scope:invested_state = {
			NOT = { owner = scope:investor_country}
		}
		OR = {
			AND = {
				NOT = { exists = scope:invested_building }
				scope:invested_state = { 
					building_livestock_ranch_total_queued < 1
				}
			}
			AND = {
				exists = scope:invested_building
				scope:invested_building = {
					is_building_type = building_livestock_ranch
					building_livestock_ranch_can_queue = yes
				}
			}
		}	
	}
	
	is_valid = {
		scope:investor_country = {
			market_capital = { owner = scope:investor_country }
		}
		scope:invested_state ={
			owner = {
				check_investable = yes
			}
		}
	}

	effect = {
		scope:investor_country = {
			set_variable = is_right_click_check
			while = {
				count = 5
				limit = { 
					has_variable = is_right_click_check
					OR = {
						AND = {
							NOT = { exists = scope:invested_building }
							scope:invested_state = { 
								building_livestock_ranch_total_queued < 1
							}
						}
						AND = {
							exists = scope:invested_building
							scope:invested_building = {
								is_building_type = building_livestock_ranch
								building_livestock_ranch_can_queue = yes
							}
						}
					}
				}
				
				
				scope:invested_state = {
					
					fi_construct_stock_exchange = {
						BUILDING = building_livestock_ranch
					}
				}
				if = {
					limit = { NOT ={ exists = scope:is_right_click } }
					remove_variable = is_right_click_check	
				}
			}
			remove_variable = is_right_click_check
		}
	}
}
