cancel_investments_coal_mine = {
	scope = country
	saved_scopes = {
		invested_state
		investor_country
		invested_building
    }

	is_shown = {
		exists = scope:invested_building
		scope:invested_building = {
			is_building_type = building_coal_mine
		}
		scope:invested_state = {
			coal_total_item > 0 
		}
	}
	
	is_valid = {
		scope:investor_country = {
			any_scope_building = {
				is_building_type = building_investment_industry
				is_under_construction = no
				state = { scope:investor_country.capital = this }
			}
		}
	}

	effect = {
		scope:invested_state = {
			fi_cancel_construction = { 
				BUILDING = building_coal_mine
			}
		}
	}
}
