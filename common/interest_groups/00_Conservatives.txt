ig_Conservatives = {
	color = rgb{ 255 182 0 }
	texture = "gfx/interface/icons/ig_icons/armed_forces.dds"
	layer = "revolution_dynamic_armed_forces"
	index = 0

	ideologies = {
		ideology_republican	
		ideology_individualist
		ideology_particularist		
	}

	enable = {
		always = yes
	}

	# Deprecated; use on_enable effect to assign traits instead
	traits = {
		ig_trait_back_Right_wing_room_dealings
		ig_trait_rational_compromise
		ig_trait_apathy
	}

	on_enable = {
		ig:ig_Conservatives = {
			if = {
				limit = {
					has_dlc_feature = voice_of_the_people_content
					owner = { c:FRA ?= this }
				}
				set_ig_trait = ig_trait:ig_trait_back_Right_wing_room_dealings
				set_ig_trait = ig_trait:ig_trait_rational_compromise
				set_ig_trait = ig_trait:ig_trait_apathy
			}
			else = {
				set_ig_trait = ig_trait:ig_trait_back_Right_wing_room_dealings
				set_ig_trait = ig_trait:ig_trait_rational_compromise
				set_ig_trait = ig_trait:ig_trait_apathy
			}
		}
	}
	on_disable = {}
	on_character_ig_membership = {}

	pop_potential = {		
		OR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
			is_pop_type = shopkeepers			
			is_pop_type = academics
			is_pop_type = clergymen
			is_pop_type = bureaucrats
			is_pop_type = clerks		
			is_pop_type = soldiers
			is_pop_type = officers
			is_pop_type = laborers
			is_pop_type = machinists
			is_pop_type = peasants
			is_pop_type = farmers
			is_pop_type = engineers			
		}
	}

	pop_weight = {
		value = 30

		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 175
			}
		}	
	
		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					strata = middle
				}
				value = 50
			}
		}

		add = {
			desc = "LOWER_CLASS"
			if = {
				limit = { 
					strata = poor
				}
				value = 30
			}
		}

		add = {
			desc = "RURAL"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 60
			}
		}

		add = {
			desc = "RURAL_UPPER_CLASS"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = rich
				}
				value = 200
			}
		}
		if = {
			limit = {
				literacy_rate = <0.05
			}
			add = {
				desc = "POP_LITERACY"
				value = literacy_rate
				multiply = 20
			}
		}
	}

	monarch_weight = {
		value = 1
		# Monarch is not likely to adopt a marginal IG
		if = {
			limit = {
				is_marginal = yes
			}
			multiply = {
				value = 0.1
			}
		}
		# Monarch is more likely to adopt a Powerful IG
		if = {
			limit = {
				is_powerful = yes
			}
			multiply = {
				value = 2
			}
		}
		# Monarch more likely to be Armed Forces under an autocratic or oligarchic regime
		if = {
			limit = {
				owner = {
					OR = {
						has_law = law_type:law_oligarchy
						has_law = law_type:law_autocracy
						has_law = law_type:law_single_party_state
					}
				}
			}
			multiply = {
				value = 2
			}
		}
		# Monarch is less likely to be Armed forces in a democratic monarchy
		if = {
			limit = {
				owner = {
					OR = {
						has_law = law_type:law_universal_suffrage
						has_law = law_type:law_census_voting
					}
				}
			}
			multiply = {
				value = 0.5
			}
		}
		# Less likely without Professional Army or Mass Conscription
		if = {
			limit = {
				owner = {
					NOR = {
						has_law = law_type:law_professional_army
						has_law = law_type:law_mass_conscription
					}
				}
			}
			multiply = {
				value = 0.5
			}
		}
	}

	agitator_weight = {
		# Agitators more likely to come from populist IG's rather than elitist
		value = 0.5
		# Agitators will never be part of a marginalized IG
		if = {
			limit = {
				owner.ig:ig_armed_forces = {
					is_marginal = yes
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	commander_weight = {
		value = 5.0
	}

	noble_chance = {
		value = 0.5
	}

	female_commander_chance = {
		value = 0.0
	}

	female_politician_chance = {
		value = 0.0
	}

	female_agitator_chance = {
		value = 0.01
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}

			}
			add = {
				value = 0.01
			}
		}	
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}

			}
			add = {
				value = 0.01
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.01
			}
		}		
	}

	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.75
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
